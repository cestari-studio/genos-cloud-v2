❌ Problemas Críticos Identificados:

#ProblemaImpactoSeveridade
1Custo fixo de 1 token por operação — não importa se é feed simples ou carrossel de 10 slidesO custo real de API (Gemini) varia de 500 a 50.000 tokens por formato. Cobrar "1" não reflete custo real.🔴 Crítico
2Sem hard-block — quando tokens/posts acabam, o sistema continua gerandoUsuário consome infinitamente. O overage acumula mas nunca bloqueia.🔴 Crítico
3Sem estimativa pré-geração — usuário não sabe quanto vai custar antes de gerarSurpresa com débito, sem transparência🟡 Alto

📊 Fórmula de conversão: API tokens → genOS tokens genOS_tokens = CEIL(api_tokens_total / 1000)

FUNCTION calculate_token_cost( p_tenant_id UUID, p_format TEXT, p_operation TEXT, p_slide_count INTEGER DEFAULT 1, p_ai_model TEXT DEFAULT 'gemini-2.0-flash' ) RETURNS INTEGER AS $$ DECLARE v_config token_cost_config%ROWTYPE; v_base INTEGER; v_slide INTEGER; v_multiplier NUMERIC; v_total INTEGER; BEGIN -- Buscar config específica do tenant, fallback para global (tenant_id IS NULL) SELECT * INTO v_config FROM token_cost_config WHERE (tenant_id = p_tenant_id OR tenant_id IS NULL) AND format = p_format AND operation = p_operation ORDER BY tenant_id NULLS LAST LIMIT 1; IF NOT FOUND THEN RETURN 1; -- fallback seguro END IF; v_base := v_config.base_cost; v_slide := COALESCE(v_config.per_slide_cost, 0) * GREATEST(p_slide_count - 1, 0); v_multiplier := COALESCE((v_config.ai_model_multiplier->>p_ai_model)::NUMERIC, 1.0); v_total := CEIL((v_base + v_slide) * v_multiplier); RETURN GREATEST(v_total, 1); END; $$ LANGUAGE plpgsql STABLE; 7. FUNCTION check_can_generate(p_tenant_id UUID) RETURNS JSONB AS $$ DECLARE v_config RECORD; v_wallet RECORD; v_usage RECORD; v_posts_this_month BIGINT; v_tokens_remaining NUMERIC; v_can_generate BOOLEAN; v_reason TEXT; BEGIN SELECT * INTO v_config FROM tenant_config WHERE tenant_id = p_tenant_id; SELECT * INTO v_wallet FROM credit_wallets WHERE tenant_id = p_tenant_id; -- Contar posts do mês corrente SELECT COUNT(*) INTO v_posts_this_month FROM posts WHERE tenant_id = p_tenant_id AND created_at >= date_trunc('month', now()); v_tokens_remaining := COALESCE(v_wallet.prepaid_credits, 0); -- Verificar hard-block IF COALESCE(v_config.hard_block_enabled, true) THEN IF v_tokens_remaining <= 0 THEN RETURN jsonb_build_object( 'allowed', false, 'reason', 'tokens_exhausted', 'message', COALESCE(v_config.zero_balance_message, 'Saldo de tokens esgotado.'), 'tokens_remaining', 0, 'posts_remaining', GREATEST(COALESCE(v_config.post_limit, 24) - v_posts_this_month, 0) ); END IF; IF v_posts_this_month >= COALESCE(v_config.post_limit, 24) THEN RETURN jsonb_build_object( 'allowed', false, 'reason', 'posts_exhausted', 'message', 'Limite de posts do mês atingido.', 'tokens_remaining', v_tokens_remaining, 'posts_remaining', 0 ); END IF; END IF; RETURN jsonb_build_object( 'allowed', true, 'tokens_remaining', v_tokens_remaining, 'posts_remaining', GREATEST(COALESCE(v_config.post_limit, 24) - v_posts_this_month, 0), 'posts_used', v_posts_this_month, 'posts_limit', COALESCE(v_config.post_limit, 24) ); END; $$ LANGUAGE plpgsql STABLE SECURITY DEFINER; 8. FUNCTION apply_addon_package(p_purchase_id UUID) RETURNS VOID AS $$ DECLARE v_purchase addon_purchases%ROWTYPE; BEGIN SELECT * INTO v_purchase FROM addon_purchases WHERE id = p_purchase_id AND status = 'approved'; IF NOT FOUND THEN RAISE EXCEPTION 'Purchase not found or not approved'; END IF; -- Creditar tokens UPDATE credit_wallets SET prepaid_credits = prepaid_credits + v_purchase.tokens_purchased, updated_at = now() WHERE tenant_id = v_purchase.tenant_id; -- Creditar posts UPDATE tenant_config SET post_limit = post_limit + v_purchase.posts_purchased WHERE tenant_id = v_purchase.tenant_id; -- Marcar como applied UPDATE addon_purchases SET status = 'applied', applied_at = now() WHERE id = p_purchase_id; -- Log INSERT INTO activity_log (tenant_id, category, title, detail) VALUES (v_purchase.tenant_id, 'commercial', 'Pacote adicional aplicado', format('+ %s tokens, + %s posts (compra #%s)', v_purchase.tokens_purchased, v_purchase.posts_purchased, p_purchase_id)); END; $$ LANGUAGE plpgsql SECURITY DEFINER; 9.
