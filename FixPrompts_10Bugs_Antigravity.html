🔴 CRITICAL BUG-01: Table Row Action Buttons Not Working

▼
src/components/ContentFactory/MatrixList.tsx
RowActions (sub-component)
~1485–1549

Diagnosis

Root Causes
  1. The OverflowMenu flipped prop may cause rendering issues in certain viewport positions
  2. Event propagation may be blocked by parent TableCell preventing click handlers from firing
  3. The post reference inside RowActions may be stale if passed incorrectly through memoization
  4. Missing e.stopPropagation() allows row click to intercept action clicks

Solution

function RowActions({ post }: { post: Post }) {
  const handleActionClick = (action: string, postId: string, e?: React.MouseEvent) => {
    if (e) {
      e.stopPropagation(); // Prevent row click
      e.preventDefault();
    }
    console.log('Action clicked:', action, postId);
    handleAction(action, postId);
  };

  return (
    <OverflowMenu size="sm" flipped={false}> // Try false first
      <OverflowMenuItem
        itemText="Visualizar"
        onClick={(e: React.MouseEvent) => {
          e.stopPropagation();
          setPreviewPost(post);
        }}
      />
      <OverflowMenuItem
        itemText="Enviar para Revisão"
        onClick={(e: React.MouseEvent) => {
          e.stopPropagation();
          handleActionClick('submit_review', post.id, e);
        }}
      />
      <OverflowMenuItem
        itemText="Aprovar"
        onClick={(e: React.MouseEvent) => {
          e.stopPropagation();
          handleActionClick('approve', post.id, e);
        }}
      />
      <OverflowMenuItem
        itemText="Solicitar Revisão"
        onClick={(e: React.MouseEvent) => {
          e.stopPropagation();
          handleActionClick('request_revision', post.id, e);
        }}
      />
      <OverflowMenuItem
        itemText="Publicar"
        onClick={(e: React.MouseEvent) => {
          e.stopPropagation();
          handleActionClick('publish', post.id, e);
        }}
      />
      <OverflowMenuItem
        itemText="Deletar"
        isDelete
        onClick={(e: React.MouseEvent) => {
          e.stopPropagation();
          handleActionClick('delete', post.id, e);
        }}
      />
    </OverflowMenu>
  );
}

🔴 CRITICAL BUG-02: BrandDna Page and Modal Don't Open

▼
src/pages/BrandDna.tsx
State initialization + dependency loops
~160–217

Diagnosis

Root Causes
  1. loadTenants useEffect has dependency [isAgency, selectedTenantId] — when selectedTenantId changes, it re-fetches ALL tenants, causing infinite loops
  2. activeTenantId = isAgency ? selectedTenantId : me.tenant?.id — for non-agency users, if me.tenant?.id is undefined initially, activeTenantId is undefined and loadData never fires
  3. loadData depends on [activeTenantId] — if undefined, the fetch never executes
  4. No fallback when me.tenant?.id is undefined during initial load

Solution

// 1. Fix loadTenants — remove selectedTenantId from dependencies:
useEffect(() => {
  const loadTenants = async () => {
    if (!isAgency || !me.tenant?.id) return;
    try {
      const { data } = await supabase
        .from('tenants')
        .select('id, name, depth_level')
        .eq('parent_tenant_id', me.tenant.id)
        .order('name');
      setAgencyTenants(data || []);
      // Auto-select first child if none selected:
      if (!selectedTenantId && data && data.length > 0) {
        setSelectedTenantId(data[0].id);
      }
    } catch (err) {
      console.error('Error loading tenants:', err);
    }
  };
  loadTenants();
}, [isAgency, me.tenant?.id]); // NOT selectedTenantId

// 2. Fix activeTenantId with fallback:
const activeTenantId = isAgency
  ? (selectedTenantId || me.tenant?.id)
  : me.tenant?.id;

// 3. Ensure loadData fires when activeTenantId changes:
useEffect(() => {
  if (activeTenantId) {
    loadData();
  }
}, [activeTenantId, loadData]);

// 4. Define loadData with proper error handling:
const loadData = useCallback(async () => {
  if (!activeTenantId) {
    console.warn('No activeTenantId, skipping data load');
    return;
  }
  setLoading(true);
  try {
    // Load current tenant's brand DNA
    const { data, error } = await supabase
      .from('brand_dna')
      .select('*')
      .eq('tenant_id', activeTenantId)
      .maybeSingle();

    if (error) throw error;

    if (data) {
      setBrandDna(data);
    } else {
      setBrandDna(null);
      showToast('Info', 'Nenhum Brand DNA configurado', 'info');
    }
  } catch (err: any) {
    console.error('Load error:', err);
    showToast('Erro', err.message, 'error');
  } finally {
    setLoading(false);
  }
}, [activeTenantId]);

🔴 CRITICAL BUG-04: Refresh Table Button Not Working

▼
src/components/ContentFactory/MatrixList.tsx
fetchPosts + Refresh Button
~695–710

Diagnosis

Root Causes
  1. fetchPosts() has concurrent fetch guard (fetchInFlight ref) and hash comparison
  2. If data hasn't changed, it silently returns without updating state
  3. When user clicks "Refresh", they expect visual feedback even if data is the same
  4. No loading indicator shows during manual refresh

Solution

// Add a forceRefresh parameter to fetchPosts:
const fetchPosts = useCallback(async (force: boolean = false) => {
  if (fetchInFlight.current && !force) return; // Allow force to bypass guard
  fetchInFlight.current = true;

  try {
    setLoading(true); // ALWAYS show loading on manual refresh

    const tenantId = isAgency ? (selectedTenantId || me.tenant?.id) : me.tenant?.id;
    if (!tenantId) return;

    const { data, error } = await supabase
      .from('posts')
      .select('*')
      .eq('tenant_id', tenantId)
      .order('created_at', { ascending: false });

    if (error) throw error;

    const newHash = JSON.stringify(data);
    // If force=true, always update; otherwise check hash
    if (force || newHash !== postsHashRef.current) {
      postsHashRef.current = newHash;
      setPosts(data || []);
      if (force) {
        showToast('Sucesso', 'Dados atualizados', 'success');
      }
    }
  } catch (err: any) {
    showToast('Erro ao carregar posts', err.message, 'error');
  } finally {
    setLoading(false);
    fetchInFlight.current = false;
  }
}, [isAgency, selectedTenantId, me.tenant?.id, showToast]);

🔴 CRITICAL BUG-06: Settings Page Changes Not Being Saved

▼
src/pages/Settings.tsx
saveConfig()
~226–285

Diagnosis

Root Causes
  1. credit_wallets update has NO error handling — if row doesn't exist, it silently fails
  2. tenant_config upsert may succeed but wallet update fails, leaving data inconsistent
  3. After save, refreshMe() is called but local config state may not reflect DB state
  4. No verification that saved values actually persisted to DB

Solution

const saveConfig = async () => {
  if (!config.tenant_id) {
    showToast('Erro', 'Nenhum tenant selecionado', 'error');
    return;
  }

  setSaving(true);
  try {
    console.log('Saving config for tenant:', config.tenant_id);

    // 1. Upsert tenant_config
    const { error: configError } = await supabase
      .from('tenant_config')
      .upsert({
        tenant_id: config.tenant_id,
        post_limit: config.post_limit,
        token_balance: config.token_balance,
        post_language: config.post_language,
        ai_model: config.ai_model,
        // ... other config fields
        updated_at: new Date().toISOString(),
      }, { onConflict: 'tenant_id' });

    if (configError) {
      console.error('Config upsert error:', configError);
      throw configError;
    }

    // 2. Update credit_wallets WITH proper error handling AND upsert:
    const { error: walletError } = await supabase
      .from('credit_wallets')
      .upsert({
        tenant_id: config.tenant_id,
        prepaid_credits: config.token_balance,
        updated_at: new Date().toISOString(),
      }, { onConflict: 'tenant_id' });

    if (walletError) {
      console.error('Wallet update failed:', walletError);
      throw walletError;
    }

    // 3. Refresh auth context to reflect changes:
    await refreshMe();

    // 4. Reload config from DB to verify it was saved:
    await loadConfig(config.tenant_id);

    showToast('Sucesso', 'Configurações salvas com sucesso', 'success');
    console.log('Config saved successfully');
  } catch (err: any) {
    console.error('Save error:', err);
    showToast('Erro ao salvar', err.message, 'error');
  } finally {
    setSaving(false);
  }
};

// Also ensure loadConfig properly loads ALL fields:

import { useEffect, useState, useCallback } from 'react';
import { useAuth } from '../contexts/AuthContext';
import { RadarChart } from '@carbon/charts-react';
import { DataTableSkeleton, Tile, Tag, Grid, Column, Section } from '@carbon/react';
import '@carbon/charts-react/styles.css';
import PageLayout from '../components/PageLayout';

interface SemanticData {
  radarData: { group: string; key: string; value: number }[];
  keywordClusters: { cluster: string; keywords: string[]; frequency: number }[];
}

export default function SemanticMapPage() {
  const { me, supabase } = useAuth();
  const [loading, setLoading] = useState(true);
  const [semanticData, setSemanticData] = useState<SemanticData | null>(null);

  const activeTenantId = me.tenant?.id;

  const fetchData = useCallback(async () => {
    if (!activeTenantId) return;
    setLoading(true);
    try {
      console.log('Semantic: fetching for tenant', activeTenantId);

      // Option A: Use edge function for AI analysis (if it exists)
      try {
        const result = await fetch('/api/semantic-analysis', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ tenant_id: activeTenantId, action: 'get_map' })
        }).then(r => r.json());

        setSemanticData(result.data || result);
        return;
      } catch (err) {
        console.log('Edge function unavailable, using fallback');
      }

      // Option B: Fallback — build from brand_dna table
      const { data: dna, error } = await supabase
        .from('brand_dna')
        .select('content_pillars, tone_attributes, brand_values, editorial_pillars')
        .eq('tenant_id', activeTenantId)
        .maybeSingle();

      if (error) throw error;

      if (dna) {
        // Transform brand_dna fields into radar chart data
        const radarData: any[] = [];

        if (dna.content_pillars) {
          dna.content_pillars.forEach((p: string, i: number) => {
            radarData.push({ group: 'Pilares', key: p, value: 60 + (i * 10) });
          });
        }

        if (dna.tone_attributes) {
          dna.tone_attributes.forEach((t: string, i: number) => {
            radarData.push({ group: 'Tom', key: t, value: 55 + (i * 8) });
          });
        }

        setSemanticData({ radarData, keywordClusters: [] });
      } else {
        setSemanticData(null);
      }
    } catch (err: any) {
      console.error('Semantic fetch error:', err);
    } finally {
      setLoading(false);
    }
  }, [activeTenantId, supabase]);

  useEffect(() => {
    fetchData();
  }, [fetchData]);

  if (loading) {
    return <PageLayout><DataTableSkeleton /></PageLayout>;
  }

  if (!semanticData) {
    return (
      <PageLayout>
        <Tile>
          <p>Configure seu Brand DNA primeiro para ver o mapa semântico.</p>
        </Tile>
      </PageLayout>
    );
  }

  return (
    <PageLayout>
      <Grid>
        <Column lg={12} md={12}>
          <Section>
            <h2>Mapa Semântico da Marca</h2>
            <RadarChart data={semanticData.radarData} options={{}} />
          </Section>
        </Column>
      </Grid>
    </PageLayout>
  );
}

    if (!isAgency || !me.tenant?.id) return;
    const { data } = await supabase
      .from('tenants')
      .select('id, name')
      .eq('parent_tenant_id', me.tenant.id)

    setAgencyTenants(data || []);

      setSelectedTenantId(data[0].id);

const activeTenantId = isAgency ? (selectedTenantId || me.tenant?.id) : me.tenant?.id;

// 2. Add error logging and fallback fetching:
const fetchData = useCallback(async () => {
  if (!activeTenantId) return;
  setLoading(true);
  try {
    console.log('QualityGate: fetching for tenant', activeTenantId);

    // Try edge functions first
    const [evalsRes, statsRes] = await Promise.allSettled([
      fetch('/api/quality-gate-evaluator', {
        method: 'POST',
        body: JSON.stringify({ action: 'list_queue', filters: { tenant_id: activeTenantId } })
      }).then(r => r.json()),
      fetch('/api/quality-gate-stats', {
        method: 'POST',
        body: JSON.stringify({ tenant_id: activeTenantId })
      }).then(r => r.json())
    ]);

    console.log('Edge results:', evalsRes, statsRes);

    if (evalsRes.status === 'fulfilled') {
      setEvaluations(evalsRes.value?.data || evalsRes.value || []);
    } else {
      console.warn('Evaluator edge function failed, using fallback');
      // Fallback to Supabase query
      const { data } = await supabase
        .from('quality_evaluations')
        .select('*, posts(title, content_type), constraint_results(*)')
        .eq('tenant_id', activeTenantId)
        .order('evaluated_at', { ascending: false)
        .limit(100);
      setEvaluations(data || []);
    }

    if (statsRes.status === 'fulfilled') {
      setStats(statsRes.value?.data || statsRes.value);
    } else {
      console.warn('Stats edge function failed, calculating from data');
      // Calculate stats from evaluations
      const evals = Array.isArray(evalsRes) ? evalsRes : [];
      if (evals.length > 0) {
        const passed = evals.filter((e) => e.status === 'passed').length;
        const failed = evals.filter((e) => e.status === 'failed').length;
        const avgScore = evals.reduce((a, e) => a + (e.overall_score || 0), 0) / evals.length;
        setStats({
          total_evaluated: evals.length,
          passed, failed,
          pass_rate: (passed / evals.length) * 100,
          avg_score: Math.round(avgScore * 100) / 100
        });
      }
    }
  } catch (err: any) {
    console.error('QualityGate fetch error:', err);
    showToast('Erro', err.message, 'error');
  } finally {
    setLoading(false);
  }
}, [activeTenantId]);
