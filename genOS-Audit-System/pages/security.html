// Edge Function com headers de segurança
export async function handler(request, context) {
  const origin = request.headers.get('origin');
  const allowedOrigins = [
    'https://app.genOS.io',
    'https://studio.genOS.io',
    'http://localhost:3000'
  ];

  const response = new Response(JSON.stringify({ data: 'content' }), {
    status: 200,
    headers: {
      'Content-Type': 'application/json',
      'Access-Control-Allow-Origin': allowedOrigins.includes(origin) ? origin : '',
      'Access-Control-Allow-Methods': 'GET,POST,PUT,DELETE,OPTIONS',
      'Access-Control-Allow-Headers': 'Content-Type,Authorization',
      'Access-Control-Max-Age': '86400',
      'X-Content-Type-Options': 'nosniff',
      'X-Frame-Options': 'DENY',
      'X-XSS-Protection': '1; mode=block',
      'Strict-Transport-Security': 'max-age=31536000; includeSubDomains',
      'Content-Security-Policy': "default-src 'self'; script-src 'self' 'unsafe-inline'; style-src 'self' 'unsafe-inline'"
    }
  });

  return response;
}

4. Implementar Token Refresh Automático

Adicionar mecanismo de renovação de JWT antes da expiração para evitar interrupção de sessão.

// React Hook para auto-refresh
import { useEffect, useRef } from 'react';
import { useAuth } from '@/context/AuthContext';

export function useTokenRefresh() {
  const { token, refreshToken } = useAuth();
  const refreshTimeoutRef = useRef(null);

  useEffect(() => {
    if (!token) return;

    // Decodificar JWT
    const decoded = JSON.parse(
      atob(token.split('.')[1])
    );
    const expiresIn = (decoded.exp * 1000) - Date.now();
    
    // Renovar 5 minutos antes de expirar
    const refreshIn = expiresIn - (5 * 60 * 1000);

    refreshTimeoutRef.current = setTimeout(async () => {
      try {
        await refreshToken();
      } catch (error) {
        console.error('Token refresh failed:', error);
        // Redirecionar para login
      }
    }, Math.max(refreshIn, 0));

    return () => {
      if (refreshTimeoutRef.current) {
        clearTimeout(refreshTimeoutRef.current);
      }
    };
  }, [token, refreshToken]);
}

// Usar no componente raiz
export function App() {
  useTokenRefresh();
  return ;
}

// Edge Function: /api/webhooks/wix/logout
import { createClient } from '@supabase/supabase-js';

export default async function handler(req, res) {
  // Validar webhook signature de Wix
  const isValid = verifyWixSignature(req);
  if (!isValid) {
    return res.status(401).json({ error: 'Unauthorized' });
  }

  const { memberId } = req.body;
  
  const supabaseAdmin = createClient(
    process.env.SUPABASE_URL,
    process.env.SUPABASE_SERVICE_ROLE_KEY
  );

  try {
    // Encontrar usuário por wix_member_id
    const { data: users } = await supabaseAdmin
      .from('profiles')
      .select('id')
      .eq('wix_member_id', memberId);

    if (users && users.length > 0) {
      const userId = users[0].id;

      // Revogar todas as sessões do usuário
      await supabaseAdmin.auth.admin.signOut(userId, 'all');

      // Log do evento
      await supabaseAdmin
        .from('audit_logs')
        .insert({
          user_id: userId,
          action: 'logout_via_wix',
          timestamp: new Date()
        });
    }

    return res.status(200).json({ success: true });
  } catch (error) {
    console.error('Logout sync error:', error);
    return res.status(500).json({ error: 'Internal server error' });
  }
}

8. Implementar Validação de app_metadata em Servidor

Adicionar middleware para validar e re-verificar metadados do usuário a cada request.

// Middleware de autenticação
import { jwtVerify } from 'jose';

export async function authMiddleware(request) {
  const authHeader = request.headers.get('authorization');
  
  if (!authHeader?.startsWith('Bearer ')) {
    return null;
  }

  const token = authHeader.slice(7);

  try {
    // Verificar assinatura do JWT
    const secret = new TextEncoder().encode(
      process.env.SUPABASE_JWT_SECRET
    );
    
    const { payload } = await jwtVerify(token, secret);

    // Validar app_metadata
    const { wix_member_id, tenant_id, role } = payload.app_metadata;
    
    if (!wix_member_id || !tenant_id || !role) {
      throw new Error('Invalid metadata');
    }

    // Re-verificar status do usuário no banco
    const { data: user, error } = await supabase
      .from('profiles')
      .select('id, is_active, role')
      .eq('wix_member_id', wix_member_id)
      .single();

    if (error || !user.is_active) {
      throw new Error('User inactive or not found');
    }

    // Atualizar role se diferente
    if (user.role !== role) {
      throw new Error('Role mismatch - token stale');
    }

    return { userId: user.id, role, tenantId: tenant_id };
  } catch (error) {
    console.error('Auth validation failed:', error);
    return null;
  }
}
